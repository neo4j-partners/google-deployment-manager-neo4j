# This is semi-automated, command line option selection will be necessary
# Docs:
# https://cloud.google.com/compute/docs/ip-addresses/reserve-static-external-ip-address

# Reserve static IP address
gcloud compute addresses create cc-core-lead-ip

# Associate address created by previous step.
STATIC_IP=35.197.104.220
gcloud compute instances add-access-config cc-core-lead \
  --access-config-name "external-static-ip" --address $STATIC_IP
